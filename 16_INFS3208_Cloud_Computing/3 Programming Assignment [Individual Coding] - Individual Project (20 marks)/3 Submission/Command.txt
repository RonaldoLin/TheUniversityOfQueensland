# Command Content

# Update package index
sudo apt -get update 

# Install Docker
sudo apt-get install docker.io

# Install Docker Compose
sudo curl -L "https://github.com/docker/compose/releases/download/1.26.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

# Create a symbolic link
sudo ln -s /usr/local/bin/docker-compose /usr/bin/docker-compose

# Confirm
Y

# Create a directory named "ip"
mkdir ip

# Navigate into the "ip" directory
cd ip

# Create a custom Docker bridge network named "my_custom_network"
sudo docker network create --driver=bridge my_custom_network

# Open the "docker-compose.yml" file for editing using the Vim text editor
vim docker-compose.yml

# Once the file is open, press "i" to enter insert mode in Vim and start editing the file.
i

# Docker Compose Content

# Define Docker Version
version: "3.3"

# Define Docker volumes
volumes:
    shared-workspace:
    hadoop_namenode:
    hadoop_datanode1:
    hadoop_datanode2:
    hadoop_datanode3:
    hadoop_historyserver:

# Define Docker network
networks:
    my_custom_network:
      external: true

# Define Docker services
services:

  # Spark Master service
  spark-master:
    image: uqteaching/cloudcomputing:spark-master-v1
    container_name: spark-master
    networks:
        - my_custom_network
    ports:
        - "8080:8080"
        - "7077:7077"
    environment:
        - INIT_DAEMON_STEP=setup_spark
        - "PYSPARK_PYTHON=/usr/bin/python3"            
        - "PYSPARK_DRIVER_PYTHON=/usr/bin/python3"            
  
  # Spark Worker 1 service
  spark-worker-1:
    image: uqteaching/cloudcomputing:spark-worker-v1
    container_name: spark-worker-1
    depends_on:
        - spark-master
    networks:
        - my_custom_network
    ports:
        - "8081:8081"
    environment:
        - "SPARK_MASTER=spark://spark-master:7077"            
        - "PYSPARK_PYTHON=/usr/bin/python3"            
        - "PYSPARK_DRIVER_PYTHON=/usr/bin/python3"            

  # Spark Worker 2 service

  # Jupyter Notebook service
  jupyter-notebook:
    container_name: jupyternb
    image: jupyter/all-spark-notebook:42f4c82a07ff
    depends_on:
        - spark-master
    expose:
        - "8888"
    networks:
        - my_custom_network
    ports:
        - "8888:8888"
    volumes:
        - ./nbs:/home/jovyan/work/nbs
        - ./events:/tmp/spark-events
    environment:
        - "PYSPARK_PYTHON=/usr/bin/python3"            
        - "PYSPARK_DRIVER_PYTHON=/usr/bin/python3"            
    command: "start-notebook.sh \
          --ip=0.0.0.0 \
          --allow-root \
          --notebook-dir=/home/jovyan/work/nbs \
          --NotebookApp.token=''"

  # Hadoop NameNode service
  namenode:
    image: bde2020/hadoop-namenode:2.0.0-hadoop3.2.1-java8
    container_name: namenode
    restart: always
    networks:
        - my_custom_network
    ports:
      - 8083:9870
      - 9000:9000
    volumes:
      - hadoop_namenode:/hadoop/dfs/name
    environment:
      - CLUSTER_NAME=test
    env_file:
      - ./Hadoop-docker/hadoop.env

  # Hadoop ResourceManager service
  resourcemanager:
    image: bde2020/hadoop-resourcemanager:2.0.0-hadoop3.2.1-java8
    container_name: resourcemanager
    restart: always
    depends_on:
